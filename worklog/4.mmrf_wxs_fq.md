input_file = sys.argv[1]
output_file = sys.argv[2]

with open(input_file, "r") as f:
    with open(output_file, "w") as out:
        header = f.readline()
        out.write(header)
        for line in f:
            fields = line.strip().split("\t")
            result_name = fields[2]
            if result_name == "readcounts_and_fpkm_tsv":
                out.write(line)
